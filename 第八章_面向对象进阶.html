第八章：面向对象进阶

8.1 Class 基础

核心概念：Class 是 ES6 引入的面向对象编程的语法糖，让类的定义和继承更加清晰。

8.1.1 类的定义

代码示例：
// 类声明
class Person {
  // 构造函数
  constructor(name, age) {
    this.name = name;
    this.age = age;
  }
  
  // 实例方法
  sayHello() {
    console.log(`Hello, I'm ${this.name}`);
  }
}

const p1 = new Person("Alice", 25);
console.log(p1);  // Person {name: "Alice", age: 25}
p1.sayHello();    // "Hello, I'm Alice"

// 类表达式
const Animal = class {
  constructor(name) {
    this.name = name;
  }
  
  speak() {
    console.log(`${this.name} makes a sound`);
  }
};

const dog = new Animal("Dog");
dog.speak();  // "Dog makes a sound"

// 类的本质是函数
console.log(typeof Person);  // "function"
console.log(Person === Person.prototype.constructor);  // true

8.1.2 类的属性

代码示例：
class Person {
  // 实例属性（公共）
  height = 1.7;
  
  // 私有属性（ES2022）
  #secret = "私有数据";
  
  // 静态属性
  static species = "Human";
  static #staticSecret = "静态私有数据";
  
  constructor(name, age) {
    this.name = name;
    this.age = age;
  }
  
  // 私有方法（ES2022）
  #privateMethod() {
    return this.#secret;
  }
  
  getSecret() {
    return this.#privateMethod();
  }
  
  // 静态方法
  static create(name, age) {
    return new Person(name, age);
  }
}

const p = new Person("Alice", 25);
console.log(p.name);        // "Alice"
console.log(p.height);      // 1.7
console.log(p.getSecret()); // "私有数据"
// console.log(p.#secret);  // SyntaxError: 私有属性不能直接访问

console.log(Person.species);  // "Human"
// console.log(Person.#staticSecret);  // SyntaxError

const p2 = Person.create("Bob", 30);
console.log(p2);  // Person {name: "Bob", age: 30, height: 1.7}

8.1.3 静态代码块

代码示例：
class Database {
  static connection;
  
  // 静态代码块（只执行一次）
  static {
    console.log("初始化数据库连接");
    Database.connection = "connected";
  }
  
  static query(sql) {
    console.log(`执行查询: ${sql}`);
    return Database.connection;
  }
}

console.log(Database.connection);  // "connected"
console.log(Database.query("SELECT * FROM users"));  // "执行查询: SELECT * FROM users"

8.2 类的继承

核心概念：使用 extends 关键字实现类的继承，super 关键字调用父类。

8.2.1 基本继承

代码示例：
// 父类
class Animal {
  constructor(name) {
    this.name = name;
  }
  
  speak() {
    console.log(`${this.name} makes a sound`);
  }
}

// 子类继承父类
class Dog extends Animal {
  constructor(name, breed) {
    super(name);  // 调用父类构造函数
    this.breed = breed;
  }
  
  // 方法重写
  speak() {
    console.log(`${this.name} barks`);
  }
  
  // 子类特有方法
  fetch() {
    console.log(`${this.name} fetches the ball`);
  }
}

const dog = new Dog("Buddy", "Golden Retriever");
dog.speak();   // "Buddy barks"
dog.fetch();   // "Buddy fetches the ball"

console.log(dog instanceof Dog);    // true
console.log(dog instanceof Animal); // true

8.2.2 super 关键字

代码示例：
class Parent {
  constructor(name) {
    this.name = name;
  }
  
  greet() {
    return `Hello, I'm ${this.name}`;
  }
}

class Child extends Parent {
  constructor(name, age) {
    super(name);  // 调用父类构造函数（必须在使用 this 之前调用）
    this.age = age;
  }
  
  greet() {
    const parentGreeting = super.greet();  // 调用父类方法
    return `${parentGreeting} and I'm ${this.age} years old`;
  }
}

const child = new Child("Alice", 25);
console.log(child.greet());  // "Hello, I'm Alice and I'm 25 years old"

8.2.3 继承内置类

代码示例：
// 继承 Array
class CustomArray extends Array {
  // 添加自定义方法
  sum() {
    return this.reduce((total, num) => total + num, 0);
  }
  
  average() {
    return this.sum() / this.length;
  }
}

const arr = new CustomArray(1, 2, 3, 4, 5);
console.log(arr.sum());      // 15
console.log(arr.average());  // 3
console.log(arr.filter(x => x > 3));  // CustomArray [4, 5]

// 继承 Map
class ObservableMap extends Map {
  constructor() {
    super();
    this.listeners = [];
  }
  
  set(key, value) {
    super.set(key, value);
    this.listeners.forEach(fn => fn(key, value));
    return this;
  }
  
  subscribe(fn) {
    this.listeners.push(fn);
  }
}

const obsMap = new ObservableMap();
obsMap.subscribe((key, value) => {
  console.log(`Map updated: ${key} = ${value}`);
});

obsMap.set("name", "Alice");  // "Map updated: name = Alice"
obsMap.set("age", 25);        // "Map updated: age = 25"

8.3 Getter 和 Setter

核心概念：使用 get 和 set 关键字定义属性的 getter 和 setter。
代码示例：
class Circle {
  constructor(radius) {
    this._radius = radius;
  }
  
  // getter
  get radius() {
    return this._radius;
  }
  
  // setter
  set radius(value) {
    if (value < 0) {
      throw new Error("半径不能为负数");
    }
    this._radius = value;
  }
  
  get area() {
    return Math.PI * this._radius ** 2;
  }
  
  get circumference() {
    return 2 * Math.PI * this._radius;
  }
}

const circle = new Circle(5);
console.log(circle.radius);         // 5
console.log(circle.area);           // 78.539...
console.log(circle.circumference);  // 31.415...

circle.radius = 10;
console.log(circle.area);  // 314.159...

// circle.radius = -1;  // Error: 半径不能为负数

8.4 Mixin 模式

核心概念：Mixin 是一种实现多重继承的方式，通过组合多个对象的功能。
代码示例：
// 定义 Mixin
const Serializable = {
  serialize() {
    return JSON.stringify(this);
  }
};

const Loggable = {
  log() {
    console.log(`[${this.constructor.name}]`, this);
  }
};

// 应用 Mixin
class User {
  constructor(name, age) {
    this.name = name;
    this.age = age;
  }
}

// 将 Mixin 的方法添加到类的原型上
Object.assign(User.prototype, Serializable, Loggable);

const user = new User("Alice", 25);
console.log(user.serialize());  // '{"name":"Alice","age":25}'
user.log();  // [User] {name: "Alice", age: 25}

// 使用函数式 Mixin
function mixin(...mixins) {
  return function(target) {
    mixins.forEach(mixin => {
      Object.getOwnPropertyNames(mixin).forEach(name => {
        target.prototype[name] = mixin[name];
      });
    });
    return target;
  };
}

const Timestamped = {
  getTimestamp() {
    return new Date().toISOString();
  }
};

@mixin(Serializable, Loggable, Timestamped)
class Message {
  constructor(content) {
    this.content = content;
  }
}

const msg = new Message("Hello");
console.log(msg.getTimestamp());  // ISO 时间字符串

8.6 本章小结

重点回顾：
  • Class 是面向对象的语法糖，本质是函数
  • 使用 constructor 定义构造函数
  • 使用 # 定义私有属性和方法
  • 使用 static 定义静态属性和方法
  • 使用 extends 实现继承，super 调用父类
  • 使用 get/set 定义属性的 getter 和 setter
  • Mixin 模式实现多重继承
