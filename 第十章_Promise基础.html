第十章：Promise 基础

10.1 异步编程的演进

核心概念：Promise 是 ES6 引入的异步编程解决方案，解决了回调地狱问题。

10.1.1 回调函数的问题

代码示例：
// 1. 回调地狱
function getUser(userId, callback) {
  setTimeout(() => {
    callback({ id: userId, name: "Alice" });
  }, 1000);
}

function getOrders(userId, callback) {
  setTimeout(() => {
    callback([{ id: 1, product: "iPhone" }]);
  }, 1000);
}

function getOrderDetails(orderId, callback) {
  setTimeout(() => {
    callback({ id: orderId, price: 999 });
  }, 1000);
}

// 回调地狱：嵌套层级深，难以维护
getUser(1, (user) => {
  console.log(user);
  getOrders(user.id, (orders) => {
    console.log(orders);
    getOrderDetails(orders[0].id, (details) => {
      console.log(details);
      // 继续嵌套...
    });
  });
});

// 2. 错误处理困难
getUser(1, (user) => {
  getOrders(user.id, (orders) => {
    getOrderDetails(orders[0].id, (details) => {
      // 每层都需要处理错误
    }, (error) => {
      console.log(error);
    });
  }, (error) => {
    console.log(error);
  });
}, (error) => {
  console.log(error);
});

10.1.2 Promise 的诞生

代码示例：
// 使用 Promise 解决回调地狱
function getUser(userId) {
  return new Promise((resolve) => {
    setTimeout(() => {
      resolve({ id: userId, name: "Alice" });
    }, 1000);
  });
}

function getOrders(userId) {
  return new Promise((resolve) => {
    setTimeout(() => {
      resolve([{ id: 1, product: "iPhone" }]);
    }, 1000);
  });
}

function getOrderDetails(orderId) {
  return new Promise((resolve) => {
    setTimeout(() => {
      resolve({ id: orderId, price: 999 });
    }, 1000);
  });
}

// 链式调用，清晰易读
getUser(1)
  .then(user => {
    console.log(user);
    return getOrders(user.id);
  })
  .then(orders => {
    console.log(orders);
    return getOrderDetails(orders[0].id);
  })
  .then(details => {
    console.log(details);
  })
  .catch(error => {
    // 统一错误处理
    console.log(error);
  });

10.2 Promise 的状态

核心概念：Promise 有三种状态：pending（等待）、fulfilled（成功）、rejected（失败）。
Promise 状态转换： ┌─────────────────────────────────────────────────────┐ │ │ │ pending (等待状态) │ │ │ │ │ ├──── resolve(value) ───→ fulfilled (成功状态) │ │ │ │ │ │ │ ↓ │ │ │ value 不可变 │ │ │ │ │ └──── reject(reason) ──→ rejected (失败状态) │ │ │ │ │ ↓ │ │ reason 不可变 │ │ │ └─────────────────────────────────────────────────────┘ 特点： 1. 状态只能从 pending 变为 fulfilled 或 rejected 2. 状态一旦确定，就不能再改变 3. 状态改变时，会调用对应的回调函数
代码示例：
// 创建 Promise
const promise = new Promise((resolve, reject) => {
  // 立即执行的函数（executor）
  console.log("executor 执行");
  
  // 模拟异步操作
  setTimeout(() => {
    const success = true;
    if (success) {
      resolve("成功的结果");  // 状态变为 fulfilled
    } else {
      reject("失败的原因");   // 状态变为 rejected
    }
  }, 1000);
});

console.log("Promise 创建完成");

// 注册回调函数
promise
  .then(value => {
    console.log("成功:", value);
  })
  .catch(error => {
    console.log("失败:", error);
  });

// 输出顺序：
// "executor 执行"
// "Promise 创建完成"
// "成功: 成功的结果"

10.2.1 状态不可变

代码示例：
// 状态一旦确定，就不能再改变
const promise = new Promise((resolve, reject) => {
  resolve("第一次 resolve");
  resolve("第二次 resolve");  // 无效
  reject("reject");           // 无效
});

promise.then(value => {
  console.log(value);  // "第一次 resolve"
});

// 在 resolve 之后抛出错误也无效
const promise2 = new Promise((resolve, reject) => {
  resolve("成功");
  throw new Error("错误");  // 无效
});

promise2.then(value => {
  console.log(value);  // "成功"
});

10.3 Promise 的基本使用

核心概念：使用 new Promise() 创建 Promise，通过 then/catch/finally 处理结果。

10.3.1 创建 Promise

代码示例：
// 基本创建
const promise = new Promise((resolve, reject) => {
  // 执行异步操作
  setTimeout(() => {
    const random = Math.random();
    if (random > 0.5) {
      resolve(random);
    } else {
      reject(new Error("随机数太小"));
    }
  }, 1000);
});

// 包装回调函数为 Promise
function setTimeoutPromise(callback, delay) {
  return new Promise((resolve) => {
    setTimeout(() => {
      resolve(callback());
    }, delay);
  });
}

setTimeoutPromise(() => "Hello", 1000)
  .then(value => console.log(value));  // "Hello"

// 封装 Ajax 请求为 Promise
function fetch(url) {
  return new Promise((resolve, reject) => {
    const xhr = new XMLHttpRequest();
    xhr.open("GET", url);
    xhr.onload = () => resolve(xhr.responseText);
    xhr.onerror = () => reject(new Error("请求失败"));
    xhr.send();
  });
}

10.3.2 处理 Promise 结果

代码示例：
const promise = new Promise((resolve, reject) => {
  setTimeout(() => {
    resolve("成功");
  }, 1000);
});

// 1. then() - 处理成功
promise.then(value => {
  console.log(value);  // "成功"
});

// 2. catch() - 处理失败
promise.catch(error => {
  console.log(error);
});

// 3. finally() - 无论成功失败都执行
promise.finally(() => {
  console.log("完成");
});

// 4. then() 可以接收两个参数
promise.then(
  value => console.log("成功:", value),
  error => console.log("失败:", error)
);

// 5. 链式调用
promise
  .then(value => {
    console.log(value);
    return "下一步";
  })
  .then(value => {
    console.log(value);
    return "再下一步";
  })
  .then(value => {
    console.log(value);
  })
  .catch(error => {
    console.log(error);
  });

10.5 本章小结

重点回顾：
  • Promise 解决了回调地狱问题
  • Promise 有三种状态：pending、fulfilled、rejected
  • 状态一旦确定，就不能再改变
  • 使用 then/catch/finally 处理 Promise 结果
  • then() 返回新的 Promise，支持链式调用
