第四章：高级技巧

4.1 深拷贝和浅拷贝

核心概念：浅拷贝只复制对象的第一层属性，深拷贝递归复制所有层级的属性。

4.1.1 浅拷贝和深拷贝的区别

浅拷贝（Shallow Copy）
只复制对象的第一层属性
嵌套对象仍然是引用
修改嵌套对象会影响原对象
深拷贝（Deep Copy）
递归复制所有层级的属性
嵌套对象也是全新的副本
修改嵌套对象不会影响原对象
代码示例：
// 浅拷贝示例
const original = {
  name: "Alice",
  age: 25,
  hobbies: ["reading", "coding"],
  info: { city: "Beijing", zip: 100000 }
};

// 使用 Object.assign（浅拷贝）
const shallowCopy = Object.assign({}, original);

// 修改基本类型属性
shallowCopy.name = "Bob";
console.log(original.name);  // "Alice"（不受影响）

// 修改嵌套对象（浅拷贝的问题）
shallowCopy.info.city = "Shanghai";
console.log(original.info.city);  // "Shanghai"（被影响了！）

// 修改嵌套数组
shallowCopy.hobbies.push("swimming");
console.log(original.hobbies);  // ["reading", "coding", "swimming"]（被影响了！）

4.1.2 浅拷贝的实现方式

1. Object.assign()
const original = { a: 1, b: { c: 2 } };
const copy = Object.assign({}, original);
console.log(copy);  // { a: 1, b: { c: 2 } }
console.log(copy.b === original.b);  // true（浅拷贝）
2. 展开运算符（Spread Operator）
const original = { a: 1, b: { c: 2 } };
const copy = { ...original };
console.log(copy);  // { a: 1, b: { c: 2 } }
console.log(copy.b === original.b);  // true（浅拷贝）

// 数组浅拷贝
const arr = [1, 2, [3, 4]];
const arrCopy = [...arr];
console.log(arrCopy[2] === arr[2]);  // true（浅拷贝）
3. Array.prototype.slice()
const arr = [1, 2, [3, 4]];
const arrCopy = arr.slice();
console.log(arrCopy);  // [1, 2, [3, 4]]
console.log(arrCopy[2] === arr[2]);  // true（浅拷贝）

4.1.3 深拷贝的实现

1. JSON.parse(JSON.stringify())
const original = {
  name: "Alice",
  age: 25,
  hobbies: ["reading", "coding"],
  info: { city: "Beijing", zip: 100000 }
};

const deepCopy = JSON.parse(JSON.stringify(original));

// 修改嵌套对象
deepCopy.info.city = "Shanghai";
console.log(original.info.city);  // "Beijing"（不受影响）

// 注意：JSON 方法的局限性
const obj = {
  date: new Date(),
  regex: /test/gi,
  func: function() {},
  undefined: undefined,
  null: null,
  nan: NaN,
  infinity: Infinity
};

const copy = JSON.parse(JSON.stringify(obj));
console.log(copy);
// {
//   date: "2024-01-01T00:00:00.000Z",  // Date 变成了字符串
//   regex: {},                          // RegExp 变成了空对象
//   // func 丢失了                      // 函数丢失
//   // undefined 丢失                   // undefined 丢失
//   null: null,
//   nan: null,                          // NaN 变成了 null
//   infinity: null                      // Infinity 变成了 null
// }
2. 递归实现深拷贝（基础版）
function deepClone(obj) {
  // 基本类型和 null 直接返回
  if (obj === null || typeof obj !== "object") {
    return obj;
  }
  
  // 创建新的对象或数组
  const clone = Array.isArray(obj) ? [] : {};
  
  // 递归复制所有属性
  for (const key in obj) {
    if (obj.hasOwnProperty(key)) {
      clone[key] = deepClone(obj[key]);
    }
  }
  
  return clone;
}

// 测试
const original = {
  name: "Alice",
  hobbies: ["reading", "coding"],
  info: { city: "Beijing" }
};

const copy = deepClone(original);
copy.info.city = "Shanghai";
console.log(original.info.city);  // "Beijing"（不受影响）
3. 完整版深拷贝（处理特殊情况）
function deepClone(obj, map = new WeakMap()) {
  // 基本类型和 null 直接返回
  if (obj === null || typeof obj !== "object") {
    return obj;
  }
  
  // 处理日期
  if (obj instanceof Date) {
    return new Date(obj);
  }
  
  // 处理正则
  if (obj instanceof RegExp) {
    return new RegExp(obj);
  }
  
  // 处理循环引用
  if (map.has(obj)) {
    return map.get(obj);
  }
  
  // 保持原型链
  const clone = Array.isArray(obj) 
    ? [] 
    : Object.create(Object.getPrototypeOf(obj));
  
  // 记录已拷贝的对象（处理循环引用）
  map.set(obj, clone);
  
  // 递归拷贝所有属性（包括 Symbol key）
  for (const key of Reflect.ownKeys(obj)) {
    const desc = Object.getOwnPropertyDescriptor(obj, key);
    if (desc.get || desc.set) {
      // 属性是 getter/setter，复制描述符
      Object.defineProperty(clone, key, desc);
    } else {
      clone[key] = deepClone(obj[key], map);
    }
  }
  
  return clone;
}

// 测试
const original = {
  name: "Alice",
  date: new Date(),
  regex: /test/gi,
  hobbies: ["reading", "coding"]
};

// 添加循环引用
original.self = original;

const copy = deepClone(original);
console.log(copy === original);          // false
console.log(copy.self === copy);         // true（循环引用正确）
console.log(copy.date === original.date); // false
console.log(copy.regex === original.regex); // false

4.1.4 深拷贝的注意事项

JSON 方法的局限性：
  • 不能处理 Date 对象（会变成字符串）
  • 不能处理 RegExp 对象（会变成空对象）
  • 不能处理函数（会丢失）
  • 不能处理 undefined（会丢失）
  • 不能处理 NaN、Infinity（会变成 null）
  • 不能处理循环引用（会报错）
  • 不能处理 Map、Set 等新数据结构
使用第三方库：
在生产环境中，建议使用成熟的第三方库，如 lodash 的 _.cloneDeep() 方法。

4.2 圣杯模式（继承）

核心概念：圣杯模式是一种实现继承的方式，可以避免共享原型的问题。

4.2.1 传统继承的问题

代码示例：
// 传统继承方式：直接赋值原型
function Animal(name) {
  this.name = name;
}

Animal.prototype.eat = function() {
  console.log(this.name + " is eating");
};

function Dog(name, breed) {
  Animal.call(this, name);
  this.breed = breed;
}

// 问题：直接赋值原型会导致共享
Dog.prototype = Animal.prototype;
Dog.prototype.constructor = Dog;

Dog.prototype.bark = function() {
  console.log(this.name + " is barking");
};

const dog = new Dog("Buddy", "Golden Retriever");
dog.bark();  // "Buddy is barking"

// 问题：修改 Dog.prototype 会影响 Animal.prototype
console.log(Animal.prototype.bark);  // function（Animal 也有 bark 方法了！）

4.2.2 圣杯模式实现

代码示例：
// 圣杯模式：使用中间对象隔离原型
function inherit(Target, Source) {
  function F() {}
  F.prototype = Source.prototype;
  Target.prototype = new F();
  Target.prototype.constructor = Target;
  Target.prototype.super = Source.prototype;
}

// 使用示例
function Animal(name) {
  this.name = name;
}

Animal.prototype.eat = function() {
  console.log(this.name + " is eating");
};

function Dog(name, breed) {
  Animal.call(this, name);
  this.breed = breed;
}

inherit(Dog, Animal);

Dog.prototype.bark = function() {
  console.log(this.name + " is barking");
};

const dog = new Dog("Buddy", "Golden Retriever");
dog.eat();   // "Buddy is eating"
dog.bark();  // "Buddy is barking"

// 不会影响 Animal.prototype
console.log(Animal.prototype.bark);  // undefined

4.2.3 圣杯模式的优化

代码示例：
// 优化：使用闭包缓存中间对象
var inherit = (function() {
  var F = function() {};
  return function(Target, Source) {
    F.prototype = Source.prototype;
    Target.prototype = new F();
    Target.prototype.constructor = Target;
    Target.prototype.super = Source.prototype;
  };
})();

// 使用示例
function Animal(name) {
  this.name = name;
}

Animal.prototype.eat = function() {
  console.log(this.name + " is eating");
};

function Dog(name, breed) {
  Animal.call(this, name);
  this.breed = breed;
}

inherit(Dog, Animal);

Dog.prototype.bark = function() {
  console.log(this.name + " is barking");
};

const dog = new Dog("Buddy", "Golden Retriever");
dog.eat();   // "Buddy is eating"
dog.bark();  // "Buddy is barking"

4.3 防抖和节流

核心概念：防抖和节流是优化高频触发事件的技术，减少函数执行次数。

4.3.1 防抖（Debounce）

防抖：在事件触发后等待一段时间，如果在这段时间内又触发了事件，则重新计时。 触发事件： ↓ ↓ ↓ ↓ ↓ ↓ 执行函数： ✓ 适用场景： - 搜索框输入联想 - 窗口大小调整 - 按钮点击防止重复提交
防抖实现：
function debounce(fn, delay, immediate = false) {
  let timer = null;
  
  return function(...args) {
    // 如果已经设置定时器，清除它
    if (timer) {
      clearTimeout(timer);
    }
    
    if (immediate) {
      // 立即执行版本：第一次触发时立即执行
      const callNow = !timer;
      timer = setTimeout(() => {
        timer = null;
      }, delay);
      if (callNow) {
        fn.apply(this, args);
      }
    } else {
      // 延迟执行版本：最后一次触发后延迟执行
      timer = setTimeout(() => {
        fn.apply(this, args);
        timer = null;
      }, delay);
    }
  };
}

// 使用示例
const searchInput = document.querySelector("#search");
const search = debounce(function(value) {
  console.log("搜索：", value);
}, 300);

searchInput.addEventListener("input", function(e) {
  search(e.target.value);
});

4.3.2 节流（Throttle）

节流：在一段时间内，无论触发多少次事件，函数只执行一次。 触发事件： ↓ ↓ ↓ ↓ ↓ ↓ 执行函数： ✓ ✓ ✓ 适用场景： - 滚动事件处理 - 鼠标移动事件 - 游戏中的射击按钮
节流实现：
function throttle(fn, delay, option = {}) {
  let timer = null;
  let previous = 0;
  
  const { leading = true, trailing = true } = option;
  
  return function(...args) {
    const now = Date.now();
    
    if (!leading && previous === 0) {
      previous = now;
    }
    
    const remaining = delay - (now - previous);
    
    if (remaining <= 0) {
      // 时间到了，执行函数
      if (timer) {
        clearTimeout(timer);
        timer = null;
      }
      fn.apply(this, args);
      previous = now;
    } else if (!timer && trailing) {
      // 设置定时器，在剩余时间后执行
      timer = setTimeout(() => {
        fn.apply(this, args);
        previous = leading ? Date.now() : 0;
        timer = null;
      }, remaining);
    }
  };
}

// 使用示例
const scrollHandler = throttle(function() {
  console.log("处理滚动事件", Date.now());
}, 1000);

window.addEventListener("scroll", scrollHandler);

4.3.3 防抖和节流的区别

防抖（Debounce）
等待一段时间后执行
如果期间又触发，重新计时
适合：搜索框、窗口调整
节流（Throttle）
固定时间间隔执行一次
无论触发多少次，只执行一次
适合：滚动事件、鼠标移动

4.3.4 实际应用场景

1. 搜索框防抖
const searchInput = document.querySelector("#search");
const searchResults = document.querySelector("#results");

const search = debounce(async function(query) {
  if (!query) {
    searchResults.innerHTML = "";
    return;
  }
  
  try {
    const response = await fetch(`/api/search?q=${query}`);
    const data = await response.json();
    searchResults.innerHTML = data.map(item => 
      `
${item.title}
`
    ).join("");
  } catch (error) {
    console.error("搜索失败：", error);
  }
}, 300);

searchInput.addEventListener("input", function(e) {
  search(e.target.value);

2. 滚动事件节流
const content = document.querySelector("#content");
let isLoading = false;

const checkScroll = throttle(function() {
  if (isLoading) return;
  
  const { scrollTop, scrollHeight, clientHeight } = content;
  
  // 检查是否滚动到底部
  if (scrollTop + clientHeight >= scrollHeight - 100) {
    isLoading = true;
    loadMoreData().finally(() => {
      isLoading = false;
    });
  }
}, 200);

content.addEventListener("scroll", checkScroll);

async function loadMoreData() {
  console.log("加载更多数据...");
  // 模拟加载
  await new Promise(resolve => setTimeout(resolve, 1000));
}
3. 按钮点击防抖
const submitButton = document.querySelector("#submit");

const submit = debounce(async function() {
  const formData = getFormData();
  
  try {
    submitButton.disabled = true;
    submitButton.textContent = "提交中...";
    
    await fetch("/api/submit", {
      method: "POST",
      body: JSON.stringify(formData)
    });
    
    alert("提交成功！");
  } catch (error) {
    alert("提交失败：" + error.message);
  } finally {
    submitButton.disabled = false;
    submitButton.textContent = "提交";
  }
}, 1000, true);  // 立即执行

submitButton.addEventListener("click", submit);

4.4 预编译和执行上下文

核心概念：JavaScript 代码执行前会进行预编译，创建执行上下文，确定变量和函数的提升。

4.4.1 执行上下文的类型

1. 全局执行上下文
// 全局执行上下文
// 1. 创建全局对象（window 或 global）
// 2. 将全局变量赋值给 window 对象的属性
// 3. 将全局函数赋值给 window 对象的属性
// 4. 将 this 指向全局对象

var globalVar = "我是全局变量";
function globalFunc() {
  console.log("我是全局函数");
}

console.log(window.globalVar);      // "我是全局变量"
console.log(window.globalFunc);     // function
console.log(this === window);       // true
2. 函数执行上下文
// 函数执行上下文
// 1. 创建活动对象（AO）
// 2. 将函数参数和变量声明赋值给 AO 对象的属性，值为 undefined
// 3. 实参和形参统一
// 4. 将函数声明赋值给 AO 对象的属性
// 5. 将 this 指向调用该函数的对象

function test(a, b) {
  console.log(a);  // function a() {}
  console.log(b);  // undefined
  
  var b = 234;
  console.log(b);  // 234
  
  a = 123;
  console.log(a);  // 123
  
  function a() {}
  var a;
  
  console.log(a);  // 123
  console.log(b);  // 234
}

test(1);

4.6 本章小结

重点回顾：
  • 浅拷贝只复制第一层属性，深拷贝递归复制所有层级
  • JSON 方法有局限性，完整深拷贝需要处理特殊情况
  • 圣杯模式使用中间对象隔离原型，避免共享问题
  • 防抖：等待一段时间后执行，期间触发重新计时
  • 节流：固定时间间隔执行一次，无论触发多少次
  • 预编译会创建执行上下文，进行变量提升和函数提升
  • 作用域链决定变量的查找顺序
