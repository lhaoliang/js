第二章：函数核心

2.1 函数对象和实例对象

核心概念：函数在 JavaScript 中既是函数也是对象，函数对象可以创建实例对象。

2.1.1 函数是对象

代码示例：
// 函数是对象的一种
function Person(name, age) {
  this.name = name;
  this.age = age;
}

// 函数对象有属性和方法
Person.prototype.sayHello = function() {
  console.log("Hello, I'm " + this.name);
};

// 函数可以添加属性
Person.species = "Human";
console.log(Person.species);  // "Human"

// 函数是 Function 的实例
console.log(Person instanceof Function);  // true
console.log(typeof Person);  // "function"
输出结果：
"Human"
true
"function"

2.1.2 实例对象

代码示例：
// 使用 new 操作符创建实例对象
function Person(name, age) {
  this.name = name;
  this.age = age;
}

Person.prototype.sayHello = function() {
  console.log("Hello, I'm " + this.name);
};

const p1 = new Person("Alice", 25);
const p2 = new Person("Bob", 30);

console.log(p1);  // Person { name: "Alice", age: 25 }
console.log(p2);  // Person { name: "Bob", age: 30 }

// 实例对象可以访问原型上的方法
p1.sayHello();  // "Hello, I'm Alice"
p2.sayHello();  // "Hello, I'm Bob"

// 实例对象是构造函数的实例
console.log(p1 instanceof Person);  // true
console.log(p2 instanceof Person);  // true
输出结果：
Person { name: "Alice", age: 25 }
Person { name: "Bob", age: 30 }
"Hello, I'm Alice"
"Hello, I'm Bob"
true
true

2.1.3 函数对象与实例对象的关系

函数对象（Constructor） 实例对象（Instance） | | | new Person("Alice", 25) | | ---------------------------> | | | | Person.prototype | __proto__ | <--------------------------- | | | Person (函数对象) ├── prototype → Person.prototype │ ├── constructor → Person │ └── sayHello() └── __proto__ → Function.prototype p1 (实例对象) ├── name: "Alice" ├── age: 25 └── __proto__ → Person.prototype ├── constructor → Person └── sayHello()

2.2 作用域和作用域链

核心概念：作用域是变量的可访问范围，作用域链是变量查找的路径。

2.2.1 作用域类型

1. 全局作用域
// 全局作用域中的变量
var globalVar = "我是全局变量";
let globalLet = "我也是全局变量";

console.log(globalVar);  // "我是全局变量"
console.log(globalLet);  // "我也是全局变量"
2. 函数作用域
function outer() {
  var outerVar = "我是外层函数的变量";
  
  function inner() {
    var innerVar = "我是内层函数的变量";
    console.log(outerVar);  // 可以访问外层变量
    console.log(innerVar);  // 可以访问内层变量
  }
  
  inner();
  // console.log(innerVar);  // 报错：innerVar is not defined
}

outer();
3. 块级作用域（ES6+）
{
  var a = "var 没有块级作用域";
  let b = "let 有块级作用域";
  const c = "const 有块级作用域";
}

console.log(a);  // "var 没有块级作用域"
// console.log(b);  // 报错：b is not defined
// console.log(c);  // 报错：c is not defined

2.2.2 作用域链

作用域链查找规则：
var name = "全局 name";

function first() {
  var name = "first 的 name";
  
  function second() {
    var name = "second 的 name";
    console.log("在 second 中:", name);  // "second 的 name"
  }
  
  second();
  console.log("在 first 中:", name);  // "first 的 name"
}

first();
console.log("在全局中:", name);  // "全局 name"
输出结果：
"在 second 中: second 的 name"
"在 first 中: first 的 name"
"在全局中: 全局 name"
作用域链查找顺序： 当前作用域 → 外层作用域 → ... → 全局作用域 示例： function outer() { var a = 10; function inner() { var b = 20; console.log(a); // 先在 inner 中找 a，没找到 // 去 outer 中找 a，找到了 console.log(c); // 先在 inner 中找 c，没找到 // 去 outer 中找 c，没找到 // 去全局中找 c，没找到 → 报错 } }

2.3 闭包

核心概念：闭包是指函数可以访问其外部作用域中的变量，即使外部函数已经执行完毕。

2.3.1 闭包的形成

代码示例：
function outer() {
  var count = 0;
  
  function inner() {
    count++;
    console.log(count);
  }
  
  return inner;
}

var counter = outer();
counter();  // 1
counter();  // 2
counter();  // 3

// 即使 outer 函数已经执行完毕，inner 函数仍然可以访问 count 变量
输出结果：
1
2
3

2.3.2 闭包的应用场景

1. 数据私有化
function createCounter() {
  let count = 0;  // 私有变量，外部无法直接访问
  
  return {
    increment() {
      count++;
      return count;
    },
    decrement() {
      count--;
      return count;
    },
    getCount() {
      return count;
    }
  };
}

const counter = createCounter();
console.log(counter.increment());  // 1
console.log(counter.increment());  // 2
console.log(counter.decrement());  // 1
console.log(counter.getCount());   // 1
// console.log(count);  // 报错：count is not defined
2. 函数柯里化
function multiply(x) {
  return function(y) {
    return x * y;
  };
}

const double = multiply(2);
const triple = multiply(3);

console.log(double(5));   // 10
console.log(double(10));  // 20
console.log(triple(5));   // 15
console.log(triple(10));  // 30
3. 记忆化（Memoization）
function memoize(fn) {
  const cache = {};
  return function(...args) {
    const key = JSON.stringify(args);
    if (cache[key]) {
      console.log("从缓存读取");
      return cache[key];
    }
    console.log("计算新值");
    const result = fn(...args);
    cache[key] = result;
    return result;
  };
}

const fibonacci = memoize(function(n) {
  return n <= 1 ? n : fibonacci(n - 1) + fibonacci(n - 2);
});

console.log(fibonacci(10));  // 计算新值 → 55
console.log(fibonacci(10));  // 从缓存读取 → 55

2.3.3 闭包的注意事项

内存泄漏风险：
闭包会保持对外部变量的引用，如果闭包长期存在，可能导致内存泄漏。
代码示例：
// 不好的示例：闭包持有大对象的引用
function createHeavyObject() {
  const bigData = new Array(1000000).fill("data");
  return function() {
    return bigData.length;
  };
}

const getLength = createHeavyObject();  // bigData 不会被垃圾回收

// 好的示例：只保留需要的数据
function createLightObject() {
  const bigData = new Array(1000000).fill("data");
  const length = bigData.length;
  return function() {
    return length;
  };
}

const getLength2 = createLightObject();  // bigData 可以被垃圾回收

2.4 闭包经典题目

核心概念：理解闭包的执行机制和变量捕获方式。

2.4.1 题目一

代码示例：
function fun(n, o) {
  console.log(o);
  return {
    fun: function(m) {
      return fun(m, n);
    }
  };
}

var a = fun(0);      // undefined
a.fun(1);            // 0
a.fun(2);            // 0
a.fun(3);            // 0

var b = fun(0)       // undefined
  .fun(1)            // 0
  .fun(2)            // 1
  .fun(3);           // 2

var c = fun(0)       // undefined
  .fun(1);           // 0
c.fun(2);            // 1
c.fun(3);            // 1
输出结果：
undefined
0
0
0
undefined
0
1
2
undefined
0
1
1
分析：
// 关键点：每次调用 fun(n, o) 都会创建新的闭包
// a.fun(1) 返回的是一个新的对象，但 a 仍然指向原来的对象
// b 是链式调用，每次 .fun() 都返回新的对象
// c 是链式调用后赋值，然后分别调用

2.4.2 题目二：循环中的闭包

代码示例：
// 经典问题：循环中的闭包
for (var i = 0; i < 5; i++) {
  setTimeout(function() {
    console.log(i);
  }, 1000);
}
// 输出：5 5 5 5 5（而不是 0 1 2 3 4）

// 原因：var 没有块级作用域，所有回调函数共享同一个 i
// 当回调函数执行时，循环已经结束，i 的值是 5

// 解决方案1：使用 let
for (let i = 0; i < 5; i++) {
  setTimeout(function() {
    console.log(i);
  }, 1000);
}
// 输出：0 1 2 3 4

// 解决方案2：使用闭包
for (var i = 0; i < 5; i++) {
  (function(j) {
    setTimeout(function() {
      console.log(j);
    }, 1000);
  })(i);
}
// 输出：0 1 2 3 4

2.4.3 题目三：闭包与变量

代码示例：
function createFunctions() {
  var result = [];
  
  for (var i = 0; i < 5; i++) {
    result.push(function() {
      return i;
    });
  }
  
  return result;
}

var functions = createFunctions();
console.log(functions[0]());  // 5
console.log(functions[1]());  // 5
console.log(functions[2]());  // 5
console.log(functions[3]());  // 5
console.log(functions[4]());  // 5

// 原因：所有函数都引用同一个变量 i
// 当函数被调用时，i 的值已经是 5

// 解决方案：使用 IIFE 或 let
function createFunctions2() {
  var result = [];
  
  for (let i = 0; i < 5; i++) {
    result.push(function() {
      return i;
    });
  }
  
  return result;
}

var functions2 = createFunctions2();
console.log(functions2[0]());  // 0
console.log(functions2[1]());  // 1
console.log(functions2[2]());  // 2
console.log(functions2[3]());  // 3
console.log(functions2[4]());  // 4

2.5 回调函数的分类

核心概念：回调函数分为同步回调和异步回调，执行时机不同。

2.5.1 同步回调

代码示例：
// 同步回调：立即执行
const arr = [1, 2, 3, 4, 5];

// forEach 是同步回调
arr.forEach(function(item) {
  console.log(item);
});
console.log("forEach 结束");

// map 是同步回调
const doubled = arr.map(function(item) {
  return item * 2;
});
console.log(doubled);

// filter 是同步回调
const evens = arr.filter(function(item) {
  return item % 2 === 0;
});
console.log(evens);
输出结果：
1
2
3
4
5
"forEach 结束"
[2, 4, 6, 8, 10]
[2, 4]

2.5.2 异步回调

代码示例：
// 异步回调：在未来的某个时间点执行
console.log("开始");

setTimeout(function() {
  console.log("setTimeout 回调");
}, 1000);

console.log("结束");

// 输出顺序：
// "开始"
// "结束"
// "setTimeout 回调"（1秒后）

// Promise 的回调也是异步的
const promise = new Promise(function(resolve) {
  console.log("Promise 构造函数");
  resolve("Promise resolve");
});

promise.then(function(value) {
  console.log(value);
});

console.log("同步代码结束");

// 输出顺序：
// "Promise 构造函数"
// "同步代码结束"
// "Promise resolve"
输出结果：
"开始"
"结束"
"setTimeout 回调"
"Promise 构造函数"
"同步代码结束"
"Promise resolve"

2.5.3 同步与异步的区别

同步回调： ┌─────────────┐ │ 调用者执行 │ │ ↓ │ │ 执行回调函数 │ ← 立即执行 │ ↓ │ │ 继续执行调用者│ └─────────────┘ 异步回调： ┌─────────────┐ │ 调用者执行 │ │ ↓ │ │ 注册回调函数 │ ← 不立即执行 │ ↓ │ │ 继续执行调用者│ │ ... │ │ 回调函数执行 │ ← 在未来某个时间点执行 └─────────────┘

2.6 函数柯里化（Currying）

核心概念：柯里化是将一个多参数函数转换为一系列单参数函数的技术。

2.6.1 柯里化的基本概念

代码示例：
// 普通函数
function add(a, b, c) {
  return a + b + c;
}
console.log(add(1, 2, 3));  // 6

// 柯里化后的函数
function curriedAdd(a) {
  return function(b) {
    return function(c) {
      return a + b + c;
    };
  };
}
console.log(curriedAdd(1)(2)(3));  // 6

// 部分应用
const add1 = curriedAdd(1);
const add12 = add1(2);
console.log(add12(3));  // 6
输出结果：
6
6
6

2.6.2 自定义 curry 函数

代码示例：
function curry(fn) {
  const arity = fn.length;  // 获取函数期望的参数个数
  
  return function curried(...args) {
    if (args.length >= arity) {
      // 参数足够，执行原函数
      return fn.call(this, ...args);
    }
    // 参数不够，返回新函数继续接收参数
    return curried.bind(this, ...args);
  };
}

// 使用示例
function multiply(a, b, c) {
  return a * b * c;
}

const curriedMultiply = curry(multiply);

console.log(curriedMultiply(2)(3)(4));    // 24
console.log(curriedMultiply(2, 3)(4));    // 24
console.log(curriedMultiply(2)(3, 4));    // 24
console.log(curriedMultiply(2, 3, 4));    // 24

// 部分应用
const double = curriedMultiply(2);
const doubleTriple = double(3);

console.log(double(5)(6));      // 2 * 5 * 6 = 60
console.log(doubleTriple(10));  // 2 * 3 * 10 = 60
输出结果：
24
24
24
24
60
60

2.6.3 柯里化的应用场景

1. 参数复用
function createUser(role, name, age) {
  return { role, name, age };
}

const curriedCreateUser = curry(createUser);

// 创建不同角色的用户工厂
const createAdmin = curriedCreateUser("admin");
const createGuest = curriedCreateUser("guest");

console.log(createAdmin("Alice", 25));
// { role: "admin", name: "Alice", age: 25 }

console.log(createGuest("Bob", 30));
// { role: "guest", name: "Bob", age: 30 }
2. 函数组合
function add(a, b) {
  return a + b;
}

function multiply(a, b) {
  return a * b;
}

const curriedAdd = curry(add);
const curriedMultiply = curry(multiply);

// 组合函数
const addTen = curriedAdd(10);
const double = curriedMultiply(2);

console.log(double(addTen(5)));  // (10 + 5) * 2 = 30
3. 日志函数
function log(level, timestamp, message) {
  console.log(`[${level}] ${timestamp}: ${message}`);
}

const curriedLog = curry(log);

// 创建特定级别的日志函数
const errorLog = curriedLog("ERROR");
const debugLog = curriedLog("DEBUG");

// 创建带时间戳的日志函数
const now = new Date().toISOString();
const errorNow = errorLog(now);

errorNow("Something went wrong!");
// [ERROR] 2024-01-01T00:00:00.000Z: Something went wrong!

debugNow("Debug message");
// [DEBUG] 2024-01-01T00:00:00.000Z: Debug message

2.8 本章小结

重点回顾：
  • 函数是对象，可以创建实例对象
  • 作用域分为全局作用域、函数作用域和块级作用域
  • 作用域链是从当前作用域到全局作用域的查找路径
  • 闭包是函数可以访问外部作用域变量的机制
  • 闭包可以用于数据私有化、柯里化、记忆化等
  • 回调函数分为同步回调和异步回调
  • 柯里化是将多参数函数转换为单参数函数的技术
