第一章：数据基础

1.1 数据类型

核心概念：JavaScript 中的数据类型分为基本数据类型、引用数据类型和其他数据类型。

1.1.1 数据类型分类

基本数据类型（Primitive Types）：
number - 数值型：任意数值（整数、浮点数、NaN、Infinity）
string - 字符串型：任意字符串
boolean - 布尔型：true 或 false
null - 空型：表示空值
undefined - 未定义型：表示变量定义了但没有赋值
symbol - 符号型：ES6新增，表示唯一值

引用数据类型（Reference Types）：
object - 对象型：任意对象
function - 函数型：一种特殊对象，可以调用
array - 数组型：一种特殊对象，数值下标，内部数据有序

其他数据类型：
bigint - 大整数型：ES2020新增，表示任意大整数

1.1.2 类型判断方法

1. typeof 操作符
typeof 变量名
返回值：字符串，表示变量的数据类型（小写）
注意：typeof 只能判断基本数据类型，不能判断引用数据类型
2. === 操作符
变量名 === 类型名
返回值：布尔值，表示变量是否是该类型
注意：可以判断 null 和 undefined
3. instanceof 操作符
变量名 instanceof 类型名
返回值：布尔值，表示变量是否是该类型的实例
注意：只能判断引用数据类型
4. Object.prototype.toString.call()
最准确的类型判断方法，可以判断所有数据类型
代码示例：
// 基本数据类型判断
let num = 123;
let str = "Hello";
let bool = true;
let empty = null;
let undef = undefined;
let sym = Symbol("test");
let big = 123n;

console.log(typeof num);        // "number"
console.log(typeof str);        // "string"
console.log(typeof bool);       // "boolean"
console.log(typeof empty);      // "object" (历史遗留问题)
console.log(typeof undef);      // "undefined"
console.log(typeof sym);        // "symbol"
console.log(typeof big);        // "bigint"

// 引用数据类型判断
let obj = {};
let arr = [];
let func = function() {};

console.log(typeof obj);        // "object"
console.log(typeof arr);        // "object"
console.log(typeof func);       // "function"

// 使用 instanceof
console.log(obj instanceof Object);   // true
console.log(arr instanceof Array);    // true
console.log(func instanceof Function); // true

// 最准确的方法：Object.prototype.toString.call()
function getType(value) {
  return Object.prototype.toString.call(value).slice(8, -1).toLowerCase();
}

console.log(getType(123));        // "number"
console.log(getType("Hello"));    // "string"
console.log(getType(true));       // "boolean"
console.log(getType(null));       // "null"
console.log(getType(undefined));  // "undefined"
console.log(getType({}));         // "object"
console.log(getType([]));         // "array"
console.log(getType(function(){})); // "function"
console.log(getType(123n));       // "bigint"
console.log(getType(Symbol()));   // "symbol"

1.2 数据、内存与变量

核心概念：数据存储在内存中，变量是内存中数据的引用。

1.2.1 基本概念

数据（Data）：
计算机中存储的信息，可以是数字、字符串、布尔值等。
内存（Memory）：
计算机中用于临时存储数据的区域，程序运行时数据存储在内存中。
变量（Variable）：
程序中用于存储数据的容器，通过 let、const 等关键字定义。

1.2.2 内存存储机制

栈内存（Stack）：
存储基本数据类型（number、string、boolean、null、undefined、symbol、bigint）
特点：大小固定，访问速度快，自动分配和释放
堆内存（Heap）：
存储引用数据类型（object、function、array）
特点：大小不固定，访问速度相对较慢，需要手动管理（垃圾回收）
代码示例：
// 基本数据类型：值存储在栈内存中
let a = 10;
let b = a;  // b 复制了 a 的值
b = 20;
console.log(a);  // 10（a 的值不变）
console.log(b);  // 20

// 引用数据类型：引用地址存储在栈内存，实际数据存储在堆内存
let obj1 = { name: "Alice", age: 25 };
let obj2 = obj1;  // obj2 复制了 obj1 的引用地址
obj2.name = "Bob";
console.log(obj1.name);  // "Bob"（obj1 也被修改了）
console.log(obj2.name);  // "Bob"

// 内存示意图：
// 栈内存          堆内存
// a: 10           
// b: 20           
// obj1: 0x123 ──→ { name: "Bob", age: 25 }
// obj2: 0x123 ──→ ↑ (同一个对象)
输出结果：
10
20
"Bob"
"Bob"

1.2.3 值传递与引用传递

值传递（Pass by Value）：
基本数据类型在赋值和函数参数传递时，传递的是值的副本。
引用传递（Pass by Reference）：
引用数据类型在赋值和函数参数传递时，传递的是引用地址的副本。
代码示例：
// 值传递示例
function addTen(num) {
  num += 10;
  console.log("函数内部：", num);  // 20
  return num;
}

let value = 10;
let result = addTen(value);
console.log("函数外部：", value);   // 10（原值不变）
console.log("返回值：", result);    // 20

// 引用传递示例
function setName(obj) {
  obj.name = "Bob";
  console.log("函数内部：", obj.name);  // "Bob"
}

let person = { name: "Alice" };
setName(person);
console.log("函数外部：", person.name);  // "Bob"（原对象被修改）

// 注意：重新赋值引用不会影响原对象
function reassign(obj) {
  obj = { name: "Charlie" };  // 创建新对象
  console.log("函数内部：", obj.name);  // "Charlie"
}

let person2 = { name: "Alice" };
reassign(person2);
console.log("函数外部：", person2.name);  // "Alice"（原对象不变）
输出结果：
函数内部： 20
函数外部： 10
返回值： 20
函数内部： "Bob"
函数外部： "Bob"
函数内部： "Charlie"
函数外部： "Alice"

1.3 数组解构

核心概念：数组解构是 ES6 引入的语法，可以快速从数组中提取值并赋值给变量。

1.3.1 基本语法

语法：
let [变量1, 变量2, 变量3, ...] = 数组;
代码示例：
// 基本解构
let [a, b, c] = [1, 2, 3];
console.log(a);  // 1
console.log(b);  // 2
console.log(c);  // 3

// 跳过元素
let [x, , z] = [1, 2, 3];
console.log(x);  // 1
console.log(z);  // 3

// 剩余元素
let [first, ...rest] = [1, 2, 3, 4, 5];
console.log(first);  // 1
console.log(rest);   // [2, 3, 4, 5]

// 默认值
let [name = "Alice", age = 25] = ["Bob"];
console.log(name);  // "Bob"
console.log(age);   // 25（使用默认值）

// 交换变量
let m = 10, n = 20;
[m, n] = [n, m];
console.log(m);  // 20
console.log(n);  // 10

1.3.2 嵌套解构

代码示例：
// 嵌套数组解构
let [a, [b, c]] = [1, [2, 3]];
console.log(a);  // 1
console.log(b);  // 2
console.log(c);  // 3

// 复杂嵌套
let [x, [y, [z]]] = [1, [2, [3]]];
console.log(x);  // 1
console.log(y);  // 2
console.log(z);  // 3
输出结果：

1.3.3 实际应用场景

1. 函数返回值解构
function getCoordinates() {
  return [10, 20, 30];
}

let [x, y, z] = getCoordinates();
console.log(x, y, z);  // 10 20 30
2. 遍历 Map
let map = new Map();
map.set("name", "Alice");
map.set("age", 25);

for (let [key, value] of map) {
  console.log(`${key}: ${value}`);
}
// 输出：
// "name: Alice"
// "age: 25"
3. 提取数组前几个元素
let colors = ["red", "green", "blue", "yellow", "purple"];

// 提取前三个
let [first, second, third] = colors;
console.log(first, second, third);  // "red" "green" "blue"

// 提取第一个和剩余
let [primary, ...others] = colors;
console.log(primary);  // "red"
console.log(others);   // ["green", "blue", "yellow", "purple"]

1.5 本章小结

重点回顾：
  • JavaScript 数据类型分为基本数据类型、引用数据类型和其他数据类型
  • typeof 只能判断基本数据类型，Object.prototype.toString.call() 最准确
  • 基本数据类型存储在栈内存，引用数据类型存储在堆内存
  • 基本数据类型是值传递，引用数据类型是引用传递
  • 数组解构可以快速提取数组中的值，支持默认值、跳过、剩余元素等语法
