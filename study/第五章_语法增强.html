第五章：语法增强

5.1 let 和 const

核心概念：let 和 const 是 ES6 引入的块级作用域变量声明方式，解决了 var 的诸多问题。

5.1.1 var 的问题

1. 没有块级作用域
// var 没有块级作用域
{
  var a = 10;
}
console.log(a);  // 10（变量泄露到外部作用域）

// 循环中的问题
for (var i = 0; i < 5; i++) {
  setTimeout(function() {
    console.log(i);
  }, 1000);
}
// 输出：5 5 5 5 5（而不是 0 1 2 3 4）
2. 变量提升
// var 会变量提升
console.log(b);  // undefined
var b = 20;
console.log(b);  // 20

// 等价于
var b;
console.log(b);  // undefined
b = 20;
console.log(b);  // 20
3. 可以重复声明
// var 可以重复声明
var c = 10;
var c = 20;
console.log(c);  // 20

5.1.2 let 的特点

var
函数作用域
会变量提升
可以重复声明
全局变量会挂载到 window
let
块级作用域
不会变量提升（暂时性死区）
不能重复声明
全局变量不会挂载到 window
代码示例：
// 1. 块级作用域
{
  let a = 10;
  console.log(a);  // 10
}
// console.log(a);  // ReferenceError: a is not defined

// 2. 暂时性死区（TDZ）
// console.log(b);  // ReferenceError: Cannot access 'b' before initialization
let b = 20;
console.log(b);  // 20

// 3. 不能重复声明
let c = 10;
// let c = 20;  // SyntaxError: Identifier 'c' has already been declared

// 4. 循环中的正确行为
for (let i = 0; i < 5; i++) {
  setTimeout(function() {
    console.log(i);
  }, 1000);
}
// 输出：0 1 2 3 4

// 5. 全局变量不会挂载到 window
let globalLet = "I'm let";
console.log(window.globalLet);  // undefined

5.1.3 const 的特点

代码示例：
// 1. 声明时必须赋值
// const a;  // SyntaxError: Missing initializer in const declaration
const a = 10;

// 2. 不能重新赋值
// a = 20;  // TypeError: Assignment to constant variable.

// 3. 块级作用域
{
  const b = 20;
  console.log(b);  // 20
}
// console.log(b);  // ReferenceError: b is not defined

// 4. 对象和数组的特殊情况
const obj = { name: "Alice" };
obj.name = "Bob";  // 可以修改对象的属性
console.log(obj);  // { name: "Bob" }

// obj = {};  // 不能重新赋值整个对象
// TypeError: Assignment to constant variable.

const arr = [1, 2, 3];
arr.push(4);  // 可以修改数组
console.log(arr);  // [1, 2, 3, 4]

// arr = [];  // 不能重新赋值整个数组
// TypeError: Assignment to constant variable.
const 的本质：
const 实际上保证的不是变量的值不得改动，而是变量指向的那个内存地址所保存的数据不得改动。 对于基本类型的数据，值就保存在变量指向的那个内存地址，因此等同于常量。 对于引用类型的数据，变量指向的内存地址保存的只是一个指向实际数据的指针， const 只能保证这个指针是固定的，至于它指向的数据结构是不是可变的，就完全不能控制了。

5.1.4 最佳实践

建议：
  • 优先使用 const，只有在需要重新赋值时才使用 let
  • 避免使用 var
  • 使用 const 声明对象和数组，但可以修改其属性和元素
代码示例：
// 推荐
const API_URL = "https://api.example.com";
const MAX_RETRIES = 3;

let currentPage = 1;
let isLoading = false;

const user = {
  name: "Alice",
  age: 25
};

const numbers = [1, 2, 3, 4, 5];

// 不推荐
var globalVar = "avoid this";

5.2 运算符扩展

核心概念：ES6 引入了许多新的运算符，简化了常见的操作。

5.2.1 展开运算符（Spread Operator）

1. 数组展开
// 数组展开
const arr1 = [1, 2, 3];
const arr2 = [4, 5, 6];

// 合并数组
const merged = [...arr1, ...arr2];
console.log(merged);  // [1, 2, 3, 4, 5, 6]

// 复制数组
const copy = [...arr1];
console.log(copy);  // [1, 2, 3]
console.log(copy === arr1);  // false（新数组）

// 函数参数
function sum(a, b, c) {
  return a + b + c;
}
const numbers = [1, 2, 3];
console.log(sum(...numbers));  // 6

// Math.max
const max = Math.max(...arr1);
console.log(max);  // 3
2. 对象展开
// 对象展开
const obj1 = { a: 1, b: 2 };
const obj2 = { c: 3, d: 4 };

// 合并对象
const merged = { ...obj1, ...obj2 };
console.log(merged);  // { a: 1, b: 2, c: 3, d: 4 }

// 复制对象
const copy = { ...obj1 };
console.log(copy);  // { a: 1, b: 2 }
console.log(copy === obj1);  // false（新对象）

// 覆盖属性
const updated = { ...obj1, b: 20 };
console.log(updated);  // { a: 1, b: 20 }

// 添加属性
const withNew = { ...obj1, e: 5 };
console.log(withNew);  // { a: 1, b: 2, e: 5 }

5.2.2 剩余参数（Rest Parameters）

代码示例：
// 剩余参数
function sum(...numbers) {
  return numbers.reduce((total, num) => total + num, 0);
}
console.log(sum(1, 2, 3, 4, 5));  // 15

// 部分参数
function greet(greeting, ...names) {
  names.forEach(name => {
    console.log(`${greeting}, ${name}!`);
  });
}
greet("Hello", "Alice", "Bob", "Charlie");
// "Hello, Alice!"
// "Hello, Bob!"
// "Hello, Charlie!"

// 解构中的剩余参数
const [first, ...rest] = [1, 2, 3, 4, 5];
console.log(first);  // 1
console.log(rest);   // [2, 3, 4, 5]

const { a, ...others } = { a: 1, b: 2, c: 3 };
console.log(a);      // 1
console.log(others); // { b: 2, c: 3 }

5.2.3 可选链操作符（Optional Chaining）

代码示例：
// 可选链操作符 ?.
const user = {
  name: "Alice",
  address: {
    city: "Beijing",
    zip: "100000"
  }
};

// 传统写法
const city1 = user && user.address && user.address.city;
console.log(city1);  // "Beijing"

// 可选链写法
const city2 = user?.address?.city;
console.log(city2);  // "Beijing"

// 访问不存在的属性
const street = user?.address?.street;
console.log(street);  // undefined

// 调用方法
const result = user?.getName?.();
console.log(result);  // undefined（方法不存在）

// 数组元素
const arr = [1, 2, 3];
const first = arr?.[0];
console.log(first);  // 1

const empty = null?.[0];
console.log(empty);  // undefined

5.2.4 空值合并操作符（Nullish Coalescing）

代码示例：
// 空值合并操作符 ??
const a = null ?? "default";
console.log(a);  // "default"

const b = undefined ?? "default";
console.log(b);  // "default"

const c = 0 ?? "default";
console.log(c);  // 0（0 不是 null 或 undefined）

const d = "" ?? "default";
console.log(d);  // ""（空字符串不是 null 或 undefined）

const e = false ?? "default";
console.log(e);  // false（false 不是 null 或 undefined）

// 与 || 的区别
const f = 0 || "default";
console.log(f);  // "default"（0 是 falsy 值）

const g = 0 ?? "default";
console.log(g);  // 0（0 不是 null 或 undefined）

5.2.5 逻辑赋值运算符

代码示例：
// 逻辑或赋值 ||=
let a = null;
a ||= "default";
console.log(a);  // "default"

let b = "existing";
b ||= "default";
console.log(b);  // "existing"（b 已经是 truthy 值）

// 逻辑与赋值 &&=
let c = "existing";
c &&= "updated";
console.log(c);  // "updated"

let d = null;
d &&= "updated";
console.log(d);  // null（d 是 falsy 值）

// 逻辑空赋值 ??=
let e = null;
e ??= "default";
console.log(e);  // "default"

let f = 0;
f ??= "default";
console.log(f);  // 0（0 不是 null 或 undefined）

5.2.6 数值分隔符

代码示例：
// 数值分隔符 _
const billion = 1_000_000_000;
console.log(billion);  // 1000000000

const bytes = 0xFF_FF_FF_FF;
console.log(bytes);  // 4294967295

const fraction = 0.123_456_789;
console.log(fraction);  // 0.123456789

const binary = 0b1010_0001_1000_0101;
console.log(binary);  // 41349

const octal = 0o1234_5670;
console.log(octal);  // 2739128

// 不能放在数字开头或结尾
// const invalid1 = _123456;  // SyntaxError

// const invalid3 = 1__234;  // SyntaxError

5.3 BigInt

核心概念：BigInt 是 ES2020 引入的新数据类型，用于表示任意精度的大整数。

5.3.1 为什么需要 BigInt

代码示例：
// Number 类型的安全整数范围
const maxInt = Number.MAX_SAFE_INTEGER;
console.log(maxInt);  // 9007199254740991

// 超出安全整数范围的计算会出错
console.log(maxInt + 1);  // 9007199254740992
console.log(maxInt + 2);  // 9007199254740992（错误！应该是 9007199254740993）

// BigInt 可以表示任意精度的整数
const bigInt = 9007199254740991n;
console.log(bigInt);  // 9007199254740991n
console.log(bigInt + 1n);  // 9007199254740992n
console.log(bigInt + 2n);  // 9007199254740993n

5.3.2 BigInt 的创建

代码示例：
// 1. 字面量创建
const big1 = 123456789012345678901234567890n;
console.log(big1);  // 123456789012345678901234567890n

// 2. BigInt() 函数创建
const big2 = BigInt(123456789012345678901234567890);
console.log(big2);  // 123456789012345678901234567890n

// 3. 从字符串创建
const big3 = BigInt("123456789012345678901234567890");
console.log(big3);  // 123456789012345678901234567890n

// 4. 从其他进制字符串创建
const big4 = BigInt("0xff");
console.log(big4);  // 255n

const big5 = BigInt("0b1010");
console.log(big5);  // 10n

const big6 = BigInt("0o777");
console.log(big6);  // 511n

5.3.3 BigInt 的运算

代码示例：
// 算术运算
const a = 100n;
const b = 200n;

console.log(a + b);   // 300n
console.log(a - b);   // -100n
console.log(a * b);   // 20000n
console.log(a / b);   // 0n（整数除法，向下取整）
console.log(a % b);   // 100n
console.log(a ** 3n); // 1000000n

// 比较运算
console.log(1n === 1);    // false（类型不同）
console.log(1n == 1);     // true（值相同）
console.log(1n < 2);      // true
console.log(2n > 1);      // true
console.log(2n >= 2);     // true

// 混合运算会报错
// console.log(1n + 1);  // TypeError: Cannot mix BigInt and other types

// 需要显式转换
console.log(1n + BigInt(1));  // 2n
console.log(Number(1n) + 1);  // 2

5.3.4 BigInt 的注意事项

注意事项：
  • BigInt 不能与 Number 混合运算
  • BigInt 转换为 Number 可能会丢失精度
  • BigInt 没有 NaN、Infinity 等特殊值
  • BigInt 除法会向下取整
  • typeof 返回 "bigint"
  • JSON.stringify 不能直接序列化 BigInt
代码示例：
// 类型检查
const big = 100n;
console.log(typeof big);  // "bigint"
console.log(big instanceof BigInt);  // false（BigInt 不是对象）

// 转换为 Number（可能丢失精度）
const bigNum = BigInt("9007199254740991");
const num = Number(bigNum);
console.log(num);  // 9007199254740991（安全范围内）

const bigNum2 = BigInt("9007199254740993");
const num2 = Number(bigNum2);
console.log(num2);  // 9007199254740992（精度丢失！）

// JSON 序列化问题
const data = { id: 100n };
// console.log(JSON.stringify(data));  // TypeError: Do not know how to serialize a BigInt

// 解决方案：自定义序列化
const dataWithBigInt = { id: 100n };
const json = JSON.stringify(dataWithBigInt, (key, value) =>
  typeof value === "bigint" ? value.toString() : value
);
console.log(json);  // {"id":"100"}

// 除法向下取整
console.log(10n / 3n);   // 3n（不是 3.333...）
console.log(10n / -3n);  // -4n（向负无穷方向取整）

// 没有 NaN
console.log(BigInt("abc"));  // SyntaxError: Cannot convert abc to a BigInt

5.3.5 BigInt 的应用场景

1. 大整数计算
// 计算大整数的阶乘
function factorial(n) {
  let result = 1n;
  for (let i = 2n; i <= BigInt(n); i++) {
    result *= i;
  }
  return result;
}

console.log(factorial(100));
// 93326215443944152681699238856266700490715968264381621468592963895217599993229915608941463976156518286253697920827223758251185210916864000000000000000000000000n

// 计算斐波那契数列
function fibonacci(n) {
  let a = 0n;
  let b = 1n;
  for (let i = 0; i < n; i++) {
    [a, b] = [b, a + b];
  }
  return a;
}

console.log(fibonacci(1000));
// 43466557686937456435688527675040625802564660517371780402481729089536555417949051890403879840079255169295922593080322634775209689623239873322471161642996440906533187938298969649928516003704476137795166849228875n
2. 精确的 ID 处理
// 数据库 ID 可能超出 Number 安全范围
const userId = 9007199254740993n;
const orderId = 9007199254740994n;

console.log(userId === orderId);  // false
console.log(userId < orderId);    // true
3. 金融计算
// 使用 BigInt 进行精确的金融计算（以分为单位）
const price = 1999n;  // 19.99 元
const quantity = 3n;
const total = price * quantity;
console.log(total);  // 5997（表示 59.97 元）

// 转换为元
const totalInYuan = Number(total) / 100;
console.log(totalInYuan);  // 59.97

5.4 globalThis

核心概念：globalThis 是 ES2020 引入的全局对象，提供了一种在任何环境下访问全局对象的标准方式。
代码示例：
// 不同环境的全局对象
// 浏览器：window
// Node.js：global
// Web Worker：self
// Service Worker：ServiceWorkerGlobalScope

// 使用 globalThis 可以在任何环境下访问全局对象
console.log(globalThis);  // 在浏览器中是 window，在 Node.js 中是 global

// 检测环境
if (globalThis.window) {
  console.log("浏览器环境");
} else if (globalThis.global) {
  console.log("Node.js 环境");
}

// 添加全局变量
globalThis.myGlobalVar = "Hello";
console.log(myGlobalVar);  // "Hello"

5.6 本章小结

重点回顾：
  • let 和 const 有块级作用域，不会变量提升
  • const 声明的变量不能重新赋值，但对象和数组的属性可以修改
  • 展开运算符可以用于数组和对象的展开、合并、复制
  • 剩余参数用于函数参数和解构赋值
  • 可选链操作符 ?. 用于安全地访问嵌套属性
  • 空值合并操作符 ?? 用于处理 null 和 undefined
  • BigInt 可以表示任意精度的大整数
  • globalThis 提供了跨环境的全局对象访问方式
