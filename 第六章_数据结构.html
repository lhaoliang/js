第六章：数据结构

6.1 Set 数据结构

核心概念：Set 是 ES6 引入的一种新的数据结构，类似于数组，但成员的值都是唯一的，没有重复的值。

6.1.1 Set 的基本操作

代码示例：
// 创建 Set
const set = new Set();
console.log(set);  // Set {}

// 添加值
set.add(1);
set.add(2);
set.add(3);
console.log(set);  // Set {1, 2, 3}

// 添加重复值（无效）
set.add(2);
console.log(set);  // Set {1, 2, 3}（没有重复）

// 使用数组初始化 Set
const set2 = new Set([1, 2, 3, 2, 1]);
console.log(set2);  // Set {1, 2, 3}

// 获取大小
console.log(set.size);  // 3

// 检查值是否存在
console.log(set.has(1));  // true
console.log(set.has(4));  // false

// 删除值
set.delete(2);
console.log(set);  // Set {1, 3}

// 清空 Set
set.clear();
console.log(set);  // Set {}
console.log(set.size);  // 0

6.1.2 Set 的遍历

代码示例：
const set = new Set(["a", "b", "c", "d"]);

// 1. for...of 循环
for (const value of set) {
  console.log(value);
}
// "a"
// "b"
// "c"
// "d"

// 2. forEach 方法
set.forEach((value, key) => {
  console.log(value, key);
});
// "a" "a"
// "b" "b"
// "c" "c"
// "d" "d"

// 3. keys() 方法
for (const key of set.keys()) {
  console.log(key);
}
// "a"
// "b"
// "c"
// "d"

// 4. values() 方法
for (const value of set.values()) {
  console.log(value);
}
// "a"
// "b"
// "c"
// "d"

// 5. entries() 方法
for (const [key, value] of set.entries()) {
  console.log(key, value);
}
// "a" "a"
// "b" "b"
// "c" "c"
// "d" "d"

6.1.3 Set 的应用场景

1. 数组去重
const arr = [1, 2, 3, 2, 1, 4, 5, 4];

// 使用 Set 去重
const unique = [...new Set(arr)];
console.log(unique);  // [1, 2, 3, 4, 5]

// 或者使用 Array.from
const unique2 = Array.from(new Set(arr));
console.log(unique2);  // [1, 2, 3, 4, 5]
2. 字符串去重
const str = "hello world";
const uniqueStr = [...new Set(str)].join("");
console.log(uniqueStr);  // "helo wrd"
3. 集合运算
const setA = new Set([1, 2, 3, 4, 5]);
const setB = new Set([4, 5, 6, 7, 8]);

// 并集
const union = new Set([...setA, ...setB]);
console.log(union);  // Set {1, 2, 3, 4, 5, 6, 7, 8}

// 交集
const intersection = new Set([...setA].filter(x => setB.has(x)));
console.log(intersection);  // Set {4, 5}

// 差集
const difference = new Set([...setA].filter(x => !setB.has(x)));
console.log(difference);  // Set {1, 2, 3}
4. 检查重复值
function hasDuplicates(arr) {
  return new Set(arr).size !== arr.length;
}

console.log(hasDuplicates([1, 2, 3, 4, 5]));  // false
console.log(hasDuplicates([1, 2, 3, 2, 5]));  // true

6.2 Map 数据结构

核心概念：Map 是 ES6 引入的一种新的数据结构，类似于对象，但键可以是任意类型的值。

6.2.1 Map 的基本操作

代码示例：
// 创建 Map
const map = new Map();
console.log(map);  // Map {}

// 添加键值对
map.set("name", "Alice");
map.set("age", 25);
console.log(map);  // Map {"name" => "Alice", "age" => 25}

// 获取值
console.log(map.get("name"));  // "Alice"
console.log(map.get("age"));   // 25
console.log(map.get("email")); // undefined

// 检查键是否存在
console.log(map.has("name"));  // true
console.log(map.has("email")); // false

// 删除键值对
map.delete("age");
console.log(map);  // Map {"name" => "Alice"}

// 获取大小
console.log(map.size);  // 1

// 清空 Map
map.clear();
console.log(map);  // Map {}
console.log(map.size);  // 0

6.2.2 Map 的键类型

代码示例：
const map = new Map();

// 字符串作为键
map.set("name", "Alice");

// 数字作为键
map.set(1, "one");

// 布尔值作为键
map.set(true, "yes");

// 对象作为键
const obj = { id: 1 };
map.set(obj, "object value");

// 数组作为键
const arr = [1, 2, 3];
map.set(arr, "array value");

// 函数作为键
const fn = function() {};
map.set(fn, "function value");

// Symbol 作为键
const sym = Symbol("test");
map.set(sym, "symbol value");

console.log(map);
// Map {
//   "name" => "Alice",
//   1 => "one",
//   true => "yes",
//   {id: 1} => "object value",
//   [1, 2, 3] => "array value",
//   function() {} => "function value",
//   Symbol(test) => "symbol value"
// }

// 注意：对象作为键时，必须是同一个引用
const obj1 = { id: 1 };
const obj2 = { id: 1 };
map.set(obj1, "obj1");
map.set(obj2, "obj2");
console.log(map.get(obj1));  // "obj1"
console.log(map.get(obj2));  // "obj2"
console.log(obj1 === obj2);  // false

6.2.3 Map 的遍历

代码示例：
const map = new Map([
  ["name", "Alice"],

// 2. forEach 方法

  console.log(`${key}: ${value}`);

// 3. keys() 方法
for (const key of map.keys()) {
  console.log(key);
}
// "name"
// "age"
// "city"

// 4. values() 方法

  console.log(value);
}
// "Alice"
// 25
// "Beijing"

// 5. entries() 方法
for (const [key, value] of map.entries()) {
  console.log(`${key}: ${value}`);
}
// "name: Alice"
// "age: 25"
// "city: Beijing"

6.2.4 Map 与 Object 的区别

Object
键只能是字符串或 Symbol
有原型链，可能有默认键
不能直接获取大小
没有内置的遍历方法
JSON 序列化支持
Map
键可以是任意类型
没有原型链，更纯净
可以通过 size 属性获取大小
有内置的遍历方法
不支持 JSON 序列化
代码示例：
// Object 的问题
const obj = {};
obj[1] = "one";
obj["1"] = "one string";
console.log(obj);  // {1: "one string"}（数字键会转换为字符串）

// Map 没有这个问题
const map = new Map();
map.set(1, "one");
map.set("1", "one string");
console.log(map.get(1));    // "one"
console.log(map.get("1"));  // "one string"

// Map 的大小
const map2 = new Map([
  ["a", 1],
  ["b", 2],
  ["c", 3]
]);
console.log(map2.size);  // 3

// Object 的大小
const obj2 = { a: 1, b: 2, c: 3 };
console.log(Object.keys(obj2).length);  // 3

6.2.5 Map 的应用场景

1. 频率统计
function countFrequency(arr) {
  const map = new Map();
  arr.forEach(item => {
    map.set(item, (map.get(item) || 0) + 1);
  });
  return map;
}

const fruits = ["apple", "banana", "apple", "orange", "banana", "apple"];
const frequency = countFrequency(fruits);
console.log(frequency);
// Map {"apple" => 3, "banana" => 2, "orange" => 1}
2. 缓存（Memoization）
function memoize(fn) {
  const cache = new Map();
  return function(...args) {
    const key = JSON.stringify(args);
    if (cache.has(key)) {
      console.log("从缓存读取");
      return cache.get(key);
    }
    console.log("计算新值");
    const result = fn(...args);
    cache.set(key, result);
    return result;
  };
}

const factorial = memoize(function(n) {
  if (n <= 1) return 1;
  return n * factorial(n - 1);
});

console.log(factorial(5));  // 计算新值 → 120
console.log(factorial(5));  // 从缓存读取 → 120
3. 双向映射
class BiMap {
  constructor() {
    this.keyToValue = new Map();
    this.valueToKey = new Map();
  }
  
  set(key, value) {
    this.keyToValue.set(key, value);
    this.valueToKey.set(value, key);
  }
  
  getByKey(key) {
    return this.keyToValue.get(key);
  }
  
  getByValue(value) {
    return this.valueToKey.get(value);
  }
}

const bimap = new BiMap();
bimap.set("name", "Alice");
bimap.set("age", 25);

console.log(bimap.getByKey("name"));    // "Alice"
console.log(bimap.getByValue("Alice")); // "name"

6.3 WeakSet 和 WeakMap

核心概念：WeakSet 和 WeakMap 是弱引用版本的 Set 和 Map，不会阻止垃圾回收。

6.3.1 WeakSet

代码示例：
// WeakSet 只能存储对象
const weakSet = new WeakSet();

const obj1 = { name: "Alice" };
const obj2 = { name: "Bob" };
const obj3 = { name: "Charlie" };

// 添加对象
weakSet.add(obj1);
weakSet.add(obj2);
console.log(weakSet.has(obj1));  // true
console.log(weakSet.has(obj2));  // true

// 删除对象
weakSet.delete(obj1);
console.log(weakSet.has(obj1));  // false

// 不能添加非对象值
// weakSet.add(1);  // TypeError: Invalid value used in weak set
// weakSet.add("hello");  // TypeError: Invalid value used in weak set

// 不能遍历
// for (const item of weakSet) {}  // TypeError: weakSet is not iterable

// 没有 size 属性
// console.log(weakSet.size);  // undefined

// 没有 clear 方法
// weakSet.clear();  // TypeError: weakSet.clear is not a function
WeakSet 的应用场景：
// 1. 存储 DOM 节点
const visitedNodes = new WeakSet();

function markVisited(node) {
  visitedNodes.add(node);
}

function isVisited(node) {
  return visitedNodes.has(node);
}

// 2. 存储已处理的对象
const processedObjects = new WeakSet();

function processObject(obj) {
  if (processedObjects.has(obj)) {
    console.log("已经处理过");
    return;
  }
  processedObjects.add(obj);
  console.log("处理对象");
}

const obj = { data: "test" };
processObject(obj);  // "处理对象"
processObject(obj);  // "已经处理过"

6.3.2 WeakMap

代码示例：
// WeakMap 的键只能是对象
const weakMap = new WeakMap();

const obj1 = { id: 1 };
const obj2 = { id: 2 };

// 添加键值对
weakMap.set(obj1, "Alice");
weakMap.set(obj2, "Bob");
console.log(weakMap.get(obj1));  // "Alice"
console.log(weakMap.get(obj2));  // "Bob"

// 检查键是否存在
console.log(weakMap.has(obj1));  // true
console.log(weakMap.has(obj2));  // true

// 删除键值对
weakMap.delete(obj1);
console.log(weakMap.has(obj1));  // false

// 不能使用非对象作为键
// weakMap.set(1, "value");  // TypeError: Invalid value used as weak map key
// weakMap.set("key", "value");  // TypeError: Invalid value used as weak map key

// 不能遍历
// for (const [key, value] of weakMap) {}  // TypeError: weakMap is not iterable

// 没有 size 属性
// console.log(weakMap.size);  // undefined

// 没有 clear 方法
// weakMap.clear();  // TypeError: weakMap.clear is not a function
WeakMap 的应用场景：
// 1. 存储对象的私有数据
const privateData = new WeakMap();

class Person {
  constructor(name, age) {
    privateData.set(this, { name, age });
  }
  
  getName() {
    return privateData.get(this).name;
  }
  
  getAge() {
    return privateData.get(this).age;
  }
}

const person = new Person("Alice", 25);
console.log(person.getName());  // "Alice"
console.log(person.getAge());   // 25

// 2. 缓存计算结果
const cache = new WeakMap();

function expensiveOperation(obj) {
  if (cache.has(obj)) {
    console.log("从缓存读取");
    return cache.get(obj);
  }
  
  console.log("计算新值");
  const result = obj.data * 2;
  cache.set(obj, result);
  return result;
}

const data = { data: 10 };
console.log(expensiveOperation(data));  // 计算新值 → 20
console.log(expensiveOperation(data));  // 从缓存读取 → 20

6.3.3 WeakSet/WeakMap 与 Set/Map 的区别

Set/Map
键/值可以是任意类型
强引用，会阻止垃圾回收
可以遍历
有 size 属性
有 clear 方法
WeakSet/WeakMap
键只能是对象
弱引用，不会阻止垃圾回收
不能遍历
没有 size 属性
没有 clear 方法
弱引用的意义：
当对象不再被其他地方引用时，垃圾回收器会自动回收该对象， WeakSet/WeakMap 中对应的条目也会自动被移除，避免内存泄漏。

6.4 Symbol

核心概念：Symbol 是 ES6 引入的一种新的原始数据类型，表示独一无二的值。

6.4.1 Symbol 的基本用法

代码示例：
// 创建 Symbol
const sym1 = Symbol();
const sym2 = Symbol();

console.log(sym1);  // Symbol()
console.log(sym2);  // Symbol()
console.log(typeof sym1);  // "symbol"

// Symbol 是唯一的
console.log(sym1 === sym2);  // false

// 带描述的 Symbol
const sym3 = Symbol("description");
console.log(sym3);  // Symbol(description)
console.log(sym3.description);  // "description"

// Symbol 不能使用 new 操作符
// const sym = new Symbol();  // TypeError: Symbol is not a constructor

6.4.2 Symbol 作为属性名

代码示例：
// Symbol 作为对象属性名
const sym = Symbol("id");

const obj = {
  [sym]: 123,
  name: "Alice"
};

console.log(obj[sym]);  // 123
console.log(obj.name);  // "Alice"

// Symbol 属性不会被 for...in 遍历
for (const key in obj) {
  console.log(key);  // 只输出 "name"
}

// Symbol 属性不会被 Object.keys() 获取
console.log(Object.keys(obj));  // ["name"]

// 获取 Symbol 属性
console.log(Object.getOwnPropertySymbols(obj));  // [Symbol(id)]

// 获取所有属性（包括 Symbol）
console.log(Reflect.ownKeys(obj));  // ["name", Symbol(id)]

6.4.3 Symbol.for() 和 Symbol.keyFor()

代码示例：
// Symbol.for() 会全局注册 Symbol
const sym1 = Symbol.for("key");
const sym2 = Symbol.for("key");

console.log(sym1 === sym2);  // true

// Symbol.keyFor() 获取全局 Symbol 的键
const sym3 = Symbol.for("test");
console.log(Symbol.keyFor(sym3));  // "test"

// 普通 Symbol 不在全局注册表中
const sym4 = Symbol("test");
console.log(Symbol.keyFor(sym4));  // undefined

6.4.4 内置 Symbol 值

代码示例：
// Symbol.iterator - 定义对象的迭代器
class Range {
  constructor(start, end) {
    this.start = start;
    this.end = end;
  }
  
  [Symbol.iterator]() {
    let current = this.start;
    const end = this.end;
    
    return {
      next() {
        if (current <= end) {
          return { value: current++, done: false };
        }
        return { done: true };
      }
    };
  }
}

const range = new Range(1, 5);
for (const num of range) {
  console.log(num);
}
// 1
// 2
// 3
// 4
// 5

// Symbol.hasInstance - 自定义 instanceof 行为
class CustomArray {
  static [Symbol.hasInstance](instance) {
    return Array.isArray(instance);
  }
}

console.log([] instanceof CustomArray);  // true
console.log({} instanceof CustomArray);  // false

// Symbol.toPrimitive - 定义对象转换为原始值的行为
class Money {
  constructor(amount) {
    this.amount = amount;
  }
  
  [Symbol.toPrimitive](hint) {
    if (hint === "number") {
      return this.amount;
    }
    if (hint === "string") {
      return `$${this.amount}`;
    }
    return this.amount;
  }
}

const money = new Money(100);
console.log(+money);      // 100
console.log(`${money}`);  // "$100"
console.log(money + 50);  // 150

6.4.5 Symbol 的应用场景

1. 避免属性名冲突
// 第三方库的属性
const LIB_ID = Symbol("lib.id");
const LIB_NAME = Symbol("lib.name");

class Library {
  constructor() {
    this[LIB_ID] = Math.random();
    this[LIB_NAME] = "My Library";
  }
  
  getId() {
    return this[LIB_ID];
  }
  
  getName() {
    return this[LIB_NAME];
  }
}

// 用户代码可以安全地添加属性
const lib = new Library();
lib.id = "user-id";
lib.name = "user-name";

console.log(lib.getId());    // 随机数
console.log(lib.id);         // "user-id"
console.log(lib.getName());  // "My Library"
console.log(lib.name);       // "user-name"
2. 定义私有属性
const _name = Symbol("name");
const _age = Symbol("age");

class Person {
  constructor(name, age) {
    this[_name] = name;
    this[_age] = age;
  }
  
  getName() {
    return this[_name];
  }
  
  getAge() {
    return this[_age];
  }
}

const person = new Person("Alice", 25);
console.log(person.getName());  // "Alice"
console.log(person.getAge());   // 25

// 无法直接访问 Symbol 属性
console.log(person._name);  // undefined
3. 定义常量
// 使用 Symbol 定义常量，确保唯一性
const DIRECTION = {
  UP: Symbol("UP"),
  DOWN: Symbol("DOWN"),
  LEFT: Symbol("LEFT"),
  RIGHT: Symbol("RIGHT")
};

function move(direction) {
  switch (direction) {
    case DIRECTION.UP:
      console.log("向上移动");
      break;
    case DIRECTION.DOWN:
      console.log("向下移动");
      break;
    case DIRECTION.LEFT:
      console.log("向左移动");
      break;
    case DIRECTION.RIGHT:
      console.log("向右移动");
      break;
  }
}

move(DIRECTION.UP);  // "向上移动"

6.6 本章小结

重点回顾：
  • Set 存储唯一值，可用于数组去重和集合运算
  • Map 存储键值对，键可以是任意类型
  • WeakSet 和 WeakMap 使用弱引用，不会阻止垃圾回收
  • Symbol 表示独一无二的值，可用于避免属性名冲突
  • Symbol.for() 会全局注册 Symbol，Symbol.keyFor() 获取全局 Symbol 的键
  • 内置 Symbol 值可以自定义对象的行为
